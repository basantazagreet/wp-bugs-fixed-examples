Bug Title / Short Description (one liner) : "WordPress is not generating thumbnails — only the original image is being saved"
Bug category: Media & File Management




Description:
A news site client noticed that their article featured images were always serving the full-resolution original — sometimes 4000×3000px images — instead of the appropriately sized thumbnails. The medium and large image sizes registered by the theme were simply not being generated on upload. Page load times had ballooned because every article was loading a 6–12MB original image.




Main Cause:
Two separate issues were confirmed:

The server had the GD extension disabled and Imagick was installed but at a version (3.4.0) that had a known bug where it silently failed to write resized images to disk without throwing any catchable PHP error. WordPress attempted resizing, received no error, but no files were created.
A poorly written plugin had added a filter to intermediate_image_sizes_advanced that was returning an empty array in all cases — effectively telling WordPress "generate zero intermediate sizes" for every upload.






What we did:



Step 1 — Diagnosed the image editor being used and tested it:

// functions.php — Temporarily log which image editor WordPress is using

add_action('admin_init', function() {
    $editor = wp_get_image_editor(
        get_attached_file(get_posts(['post_type' => 'attachment', 'numberposts' => 1])[0]->ID)
    );
    if (is_wp_error($editor)) {
        error_log('Image editor error: ' . $editor->get_error_message());
    } else {
        error_log('Image editor class: ' . get_class($editor));

        // Test a resize operation directly
        $result = $editor->resize(300, 200, true);
        error_log('Resize test: ' . (is_wp_error($result) ? $result->get_error_message() : 'OK'));

        $saved = $editor->save('/tmp/test-resize.jpg');
        error_log('Save test: ' . (is_wp_error($saved) ? $saved->get_error_message() : 'OK - ' . $saved['path']));
    }
});






Step 2 — Forced WordPress to use GD as the image editor (after confirming GD was available):

// functions.php — Force GD image editor over Imagick
add_filter('wp_image_editors', function($editors) {
    // Move GD to the front of the preference list
    return ['WP_Image_Editor_GD', 'WP_Image_Editor_Imagick'];
});






Step 3 — Found and removed the broken intermediate_image_sizes_advanced filter:


// Debug — check what's filtering intermediate image sizes
add_filter('intermediate_image_sizes_advanced', function($sizes, $metadata, $attachment_id) {
    error_log('Intermediate sizes requested: ' . print_r(array_keys($sizes), true));
    error_log('Sizes after filters: ' . print_r(array_keys(apply_filters(
        'intermediate_image_sizes_advanced', $sizes, $metadata, $attachment_id
    )), true));
    return $sizes;
}, 1, 3); // Priority 1 = runs before other filters

// Log revealed a plugin was returning [] — found it via plugin deactivation
// Removed the offending filter:
remove_filter('intermediate_image_sizes_advanced', ['BrokenPlugin', 'filter_image_sizes'], 10);





Step 4 — Verified and registered all required image sizes for the theme:

// functions.php — Ensure all required image sizes are registered
add_action('after_setup_theme', function() {
    add_theme_support('post-thumbnails');

    // Core sizes
    update_option('thumbnail_size_w', 150);
    update_option('thumbnail_size_h', 150);
    update_option('thumbnail_crop',   1);

    update_option('medium_size_w',    768);
    update_option('medium_size_h',    0);   // 0 = proportional height

    update_option('large_size_w',     1200);
    update_option('large_size_h',     0);

    // Custom news site sizes
    add_image_size('news-featured',   1200, 630,  true);  // OG image ratio
    add_image_size('news-card',       600,  400,  true);  // Article card
    add_image_size('news-thumbnail',  300,  200,  true);  // List thumbnail
    add_image_size('news-wide',       1920, 600,  true);  // Hero banner
});





Step 5 — Regenerated all missing image sizes for existing attachments:


# Regenerate all missing intermediate sizes for all attachments

wp media regenerate --yes --skip-delete

# Verify sizes are now being generated for a specific attachment

wp eval '
$att_id = 1234; // Replace with a real attachment ID
$meta = wp_get_attachment_metadata($att_id);
echo "Sizes available:\n";
print_r(array_keys($meta["sizes"]));
'





Step 6 — Added an upload-time size verification to catch future generation failures early:

// functions.php — Verify expected sizes were generated after each upload
add_action('add_attachment', function($attachment_id) {
    // Wait for metadata to be written
    $meta           = wp_get_attachment_metadata($attachment_id);
    $expected_sizes = ['thumbnail', 'medium', 'large', 'news-featured', 'news-card'];
    $missing        = [];

    if (!isset($meta['sizes'])) {
        error_log("No intermediate sizes generated for attachment $attachment_id");
        return;
    }

    foreach ($expected_sizes as $size) {
        if (!isset($meta['sizes'][$size])) {
            $missing[] = $size;
        }
    }

    if (!empty($missing)) {
        error_log(
            "Attachment $attachment_id missing sizes: " . implode(', ', $missing)
        );
    }
});







After regeneration, all existing images had their correct intermediate sizes. New uploads started generating all 7 registered sizes correctly. Page load times dropped by 76% on article pages.
